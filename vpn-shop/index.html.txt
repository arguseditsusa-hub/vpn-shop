        // Инициализируем WebApp
        const tg = window.Telegram.WebApp;
        tg.expand(); // растягиваем на всю высоту

        // Обработчики кликов по тарифам
        document.querySelectorAll('.tariff').forEach(el => {
            el.addEventListener('click', function() {
                const tariff = this.dataset.tariff;
                // Отправляем данные в бота
                tg.sendData(JSON.stringify({ tariff: tariff }));
                document.getElementById('status').textContent = '⏳ Отправляем запрос...';
                // Закрываем WebApp после отправки (можно через секунду)
                setTimeout(() => tg.close(), 1000);
            });
